/*弗洛伊德算法*/
int floyd() {
    int i = 1, j = 1, k = 1, l = 1;
    for (i = 1; i <= Num; i++) {
        for (j = 1; j <= Num; j++) {
            shortest[i][j] = edge[i][j];
            path[i][j] = 0;
        }
    }
    for (k = 1; k <= Num; k++) {
        for (i = 1; i <= Num; i++) {
            for (j = 1; j <= Num; j++) {
                if (shortest[i][j] > (shortest[i][k] + shortest[k][j])) {
                    shortest[i][j] = (shortest[i][k] + shortest[k][j]);
                    path[i][j] = path[j][i] = k;
                }
            }
        }
    }
    return 0;
}
